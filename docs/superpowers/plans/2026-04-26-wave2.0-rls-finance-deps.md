# Wave 2.0 Implementation Plan — RLS Hardening, Finance Role, Dep Cleanup

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship Wave 2.0 of the JPR critical-gaps fix: enable Postgres row-level security on every sensitive table with `withRLS()` wired through every action and API route, add a `finance` role with its own route group, and remove unused dependencies. Wave 2.1 (feature work) ships separately after Wave 2.0 has soaked for one full payroll/invoicing cycle.

**Architecture:** RLS hybrid — DB-level `CREATE POLICY` for sensitive tables (with `FORCE ROW LEVEL SECURITY` because `jpr_migrator` owns tables), `withRLS()` invoked from every action/API route that touches them, app-layer role checks remain for non-sensitive tables. Two Postgres roles: `jpr_app` (NOBYPASSRLS, runtime) and `jpr_migrator` (BYPASSRLS, DDL/seed). One `system@internal` user with login disabled drives cron paths under `super_admin` via `withRLS(SYSTEM_USER_ID, 'super_admin', …)`. New `finance` role gets a `/finance/*` route group; admin/super_admin can also reach it.

**Tech Stack:** Next.js 15 (App Router), Drizzle ORM, node-postgres `pg`, PostgreSQL ≥ 14, jose JWT, Vitest, ESLint 9, next-intl, bcryptjs.

**Spec:** [docs/superpowers/specs/2026-04-26-wave2-critical-gaps-design.md](../specs/2026-04-26-wave2-critical-gaps-design.md) — Section 2.

import { sql } from "drizzle-orm";
import { db } from "./index";

let asserted = false;

export async function assertRlsPosture(): Promise<void> {
  if (asserted) return;
  if (process.env.NODE_ENV === "test") {
    asserted = true;
    return;
  }

  const role = await db.execute<{
    rolbypassrls: boolean;
    rolsuper: boolean;
  }>(sql`SELECT rolbypassrls, rolsuper FROM pg_roles WHERE rolname = current_user`);
  const r = (role.rows ?? role)[0] as
    | { rolbypassrls: boolean; rolsuper: boolean }
    | undefined;

  if (!r) {
    throw new Error("[rls] could not read pg_roles for current_user");
  }
  if (r.rolsuper) {
    throw new Error(
      "[rls] runtime DB role is SUPERUSER — refusing to start. Use jpr_app, not the migrator role."
    );
  }
  if (r.rolbypassrls) {
    throw new Error(
      "[rls] runtime DB role has BYPASSRLS — refusing to start. Use jpr_app, not the migrator role."
    );
  }

  // Fail-closed probe — sensitive table without withRLS must return 0 rows.
  // We pick daily_logs because it has policies installed in this PR.
  const probe = await db.execute<{ count: string }>(
    sql`SELECT count(*)::text AS count FROM daily_logs`
  );
  const count = Number((probe.rows ?? probe)[0]?.count ?? "0");
  if (count !== 0) {
    throw new Error(
      `[rls] fail-closed probe failed: SELECT count(*) FROM daily_logs returned ${count} without withRLS — RLS not enforced`
    );
  }

  asserted = true;
}

import { beforeAll } from "vitest";
import { Pool } from "pg";
import { drizzle } from "drizzle-orm/node-postgres";
import { sql } from "drizzle-orm";
import * as schema from "@/db/schema";

export const appPool = new Pool({
  connectionString: process.env.DATABASE_URL,
  max: 4,
});
export const migratorPool = new Pool({
  connectionString: process.env.MIGRATION_DATABASE_URL ?? process.env.DATABASE_URL,
  max: 2,
});

export const appDb = drizzle(appPool, { schema });
export const migratorDb = drizzle(migratorPool, { schema });

beforeAll(async () => {
  if (!process.env.DATABASE_URL) {
    throw new Error("DATABASE_URL must be set for RLS integration tests");
  }
  // Sanity: ensure runtime role does NOT have BYPASSRLS.
  const r = await appPool.query<{ rolbypassrls: boolean; rolsuper: boolean }>(
    "SELECT rolbypassrls, rolsuper FROM pg_roles WHERE rolname = current_user"
  );
  if (r.rows[0]?.rolsuper || r.rows[0]?.rolbypassrls) {
    throw new Error(
      "RLS tests require DATABASE_URL to be the jpr_app role (NOSUPERUSER NOBYPASSRLS)"
    );
  }
});

export async function asRole<T>(
  userId: string,
  role: string,
  fn: (tx: typeof appDb) => Promise<T>
): Promise<T> {
  return appDb.transaction(async (tx) => {
    await tx.execute(sql`SET LOCAL app.current_user_id = ${userId}`);
    await tx.execute(sql`SET LOCAL app.current_user_role = ${role}`);
    return fn(tx as unknown as typeof appDb);
  });
}

export async function unwrapped<T>(
  fn: (db: typeof appDb) => Promise<T>
): Promise<T> {
  return fn(appDb);
}

// Seed helpers
export async function seedUserAndStaff(opts: {
  phone: string;
  role: string;
}): Promise<{ userId: string; staffId: string | null }> {
  const u = await migratorPool.query<{ id: string }>(
    `INSERT INTO users(phone, password_hash, role, preferred_locale, is_active)
     VALUES ($1, '$2b$10$test', $2, 'en', true)
     ON CONFLICT (phone) DO UPDATE SET role = EXCLUDED.role
     RETURNING id`,
    [opts.phone, opts.role]
  );
  const userId = u.rows[0].id;

  let staffId: string | null = null;
  if (opts.role === "operator") {
    const s = await migratorPool.query<{ id: string }>(
      `INSERT INTO staff_profiles(user_id, full_name, phone, pay_rate, pay_type)
       VALUES ($1, $2, $3, 1000, 'daily')
       ON CONFLICT DO NOTHING
       RETURNING id`,
      [userId, opts.phone, opts.phone]
    );
    staffId = s.rows[0]?.id ?? null;
    if (!staffId) {
      const e = await migratorPool.query<{ id: string }>(
        `SELECT id FROM staff_profiles WHERE user_id = $1`,
        [userId]
      );
      staffId = e.rows[0].id;
    }
  }

  return { userId, staffId };
}

import { describe, it, expect, beforeAll } from "vitest";
import { sql } from "drizzle-orm";
import { asRole, unwrapped, migratorPool, seedUserAndStaff } from "./setup";

describe("daily_logs RLS", () => {
  let aliceUser: string;
  let aliceStaff: string;
  let bobUser: string;
  let bobStaff: string;
  let aliceLogId: string;
  let bobLogId: string;

      role: "operator",
    })) as any);
    ({ userId: bobUser, staffId: bobStaff! } = (await seedUserAndStaff({
      phone: "0770000011",
      role: "operator",
    })) as any);

    // Insert two logs as the migrator (BYPASSRLS) — one for each operator.
    const a = await migratorPool.query<{ id: string }>(
      `INSERT INTO daily_logs(operator_id, vehicle_id, date, start_time, end_time, start_engine_hours, end_engine_hours, sync_status)
       VALUES ($1, gen_random_uuid(), CURRENT_DATE, NOW(), NOW(), 0, 1, 'synced') RETURNING id`,
      [aliceStaff]
    );
    aliceLogId = a.rows[0].id;
    const b = await migratorPool.query<{ id: string }>(
      `INSERT INTO daily_logs(operator_id, vehicle_id, date, start_time, end_time, start_engine_hours, end_engine_hours, sync_status)
       VALUES ($1, gen_random_uuid(), CURRENT_DATE, NOW(), NOW(), 0, 1, 'synced') RETURNING id`,
      [bobStaff]
    );
    bobLogId = b.rows[0].id;
  });

  it("operator sees own log only", async () => {
    const rows = await asRole(aliceUser, "operator", (tx) =>
      tx.execute<{ id: string }>(sql`SELECT id FROM daily_logs ORDER BY id`)
    );
    const ids = (rows.rows ?? rows).map((r: any) => r.id);
    expect(ids).toContain(aliceLogId);
    expect(ids).not.toContain(bobLogId);
  });

  it("admin sees all logs", async () => {
    const adminUser = (await seedUserAndStaff({
      phone: "0770000020",
      role: "admin",
    })).userId;
    const rows = await asRole(adminUser, "admin", (tx) =>
      tx.execute<{ id: string }>(sql`SELECT id FROM daily_logs ORDER BY id`)
    );
    const ids = (rows.rows ?? rows).map((r: any) => r.id);
    expect(ids).toContain(aliceLogId);
    expect(ids).toContain(bobLogId);
  });

  it("auditor sees all logs but cannot UPDATE", async () => {
    const auditorUser = (await seedUserAndStaff({
      phone: "0770000021",
      role: "auditor",
    })).userId;
    const updated = await asRole(auditorUser, "auditor", (tx) =>
      tx.execute(
        sql`UPDATE daily_logs SET notes = 'auditor-attempt' WHERE id = ${aliceLogId} RETURNING id`
      )
    );
    expect((updated.rows ?? updated).length).toBe(0);
  });

  it("operator cannot insert with another operator's id", async () => {
    await expect(

        `)

    ).rejects.toThrow(/policy/);

    const rows = await unwrapped((db) =>
      db.execute<{ count: string }>(sql`SELECT count(*)::text AS count FROM daily_logs`)
    );
    const count = Number((rows.rows ?? rows)[0]?.count ?? "0");
    expect(count).toBe(0);
  });
});
